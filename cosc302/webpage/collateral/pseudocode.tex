\noindent{\bf \LaTeX{} Example: Pseudocode for insertion sort} \\

\begin{algorithm}[!h]
\textsc{Insertion-Sort}($A$)
\begin{algorithmic}[1]
\For{$j = 2$ \textbf{to} $A$.length}
\State $key = A[j]$
\State $i = j-1$
\While{$i > 0$ and $A[i] > key$}
\State $A[i+1] = A[i]$
\State $i = i-1$
\EndWhile
\State $A[i+1] = key$
\EndFor
\end{algorithmic}
\end{algorithm}
